// ─── Image Fallback & Search Bar ─────────────────────────────────────────────

// Called from img onerror attribute in HTML
function handleImgError(img) {
  img.onerror = null;
  img.style.display = 'none';

  const fallback = $('photo-fallback');
  fallback.classList.remove('hidden');
  if (state.currentPerson) {
    const words    = state.currentPerson.name.split(' ').filter(w => /[A-Za-z]/.test(w[0]));
    const initials = words.slice(0, 2).map(w => w[0].toUpperCase()).join('');
    $('fallback-emoji').textContent    = state.currentPerson.emoji;
    $('fallback-initials').textContent = initials;
  }

  showPhotoSearch();
}

function showPhotoSearch() {
  const bar = $('photo-search');
  if (!bar) return;
  bar.classList.remove('hidden');
  const person = state.currentPerson;
  if (person) {
    const birthYear = person.years.split(/[–—-]/)[0].trim();
    $('photo-search-input').value = person.name + ' ' + birthYear;
  }
  $('photo-search-status').textContent = '';
}

function hidePhotoSearch() {
  const bar = $('photo-search');
  if (bar) bar.classList.add('hidden');
}

async function fetchWikiPhoto(query) {
  const slug = query.trim().replace(/\s+/g, '_');
  const url  = `https://en.wikipedia.org/api/rest_v1/page/summary/${encodeURIComponent(slug)}`;
  const res  = await fetch(url);
  if (!res.ok) return null;
  const data = await res.json();
  if (data.thumbnail && data.thumbnail.source) {
    return data.thumbnail.source.replace(/\/\d+px-([^/]+)$/, '/600px-$1');
  }
  return null;
}

async function handlePhotoSearch(e) {
  e.preventDefault();
  const query  = $('photo-search-input').value.trim();
  const status = $('photo-search-status');
  const btn    = $('photo-search-btn');
  if (!query) return;

  btn.disabled = true;
  btn.textContent = '…';
  status.textContent = 'Searching…';

  try {
    const photoUrl = await fetchWikiPhoto(query);
    if (photoUrl) {
      const img     = $('person-img');
      const testImg = new Image();
      testImg.referrerPolicy = 'no-referrer';
      testImg.onload = () => {
        img.style.display = '';
        $('photo-fallback').classList.add('hidden');
        img.src = photoUrl;
        status.textContent = 'Found!';
        setTimeout(hidePhotoSearch, 800);
      };
      testImg.onerror = () => {
        status.textContent = 'Image didn\'t load — try a different search.';
        btn.disabled    = false;
        btn.textContent = 'Search';
      };
      testImg.src = photoUrl;
    } else {
      status.textContent = 'No photo found — try a different search.';
      btn.disabled    = false;
      btn.textContent = 'Search';
    }
  } catch (_) {
    status.textContent = 'Search failed — check your connection.';
    btn.disabled    = false;
    btn.textContent = 'Search';
  }
}

// ─── Toggle Buttons ───────────────────────────────────────────────────────────

function updateToggles() {
  document.querySelectorAll('.toggle-btn').forEach(btn => {
    const turn = parseInt(btn.dataset.turn);
    const idx  = btn.dataset.type === 'comment' ? 0 : 1;
    const v    = state.turns[turn][idx]; // null | 'check' | 'ok'
    btn.textContent = v === 'check' ? '✓' : v === 'ok' ? 'OK' : '—';
    btn.classList.toggle('checked', v === 'check');
    btn.classList.toggle('ok',      v === 'ok');
  });
}

function handleToggleClick(e) {
  const btn = e.target.closest('.toggle-btn');
  if (!btn) return;
  const turn = parseInt(btn.dataset.turn);
  const idx  = btn.dataset.type === 'comment' ? 0 : 1;

  if (state.turns[turn][idx] !== null) {
    // Un-mark: reset cell tracking and clear
    state.turns[turn][idx] = null;
    state.ltmCellLevels[turn][idx]      = 0;
    state.promptClickedForCell[turn][idx] = false;
    clearTimeout(autoShowTimeout);
    updateToggles();
    displayPrompt('');
    return;
  }

  // Determine ✓ or OK
  let result;
  if (state.promptMethod === 'mtl') {
    // Clear any pending delay timeout — client responded on their own
    clearTimeout(autoShowTimeout);
    result = state.promptClickedForCell[turn][idx] ? 'ok' : 'check';
  } else {
    // LtM: compare cell's reached level to prescribed ceiling
    result = state.ltmCellLevels[turn][idx] > state.mtlLevel ? 'ok' : 'check';
  }

  state.turns[turn][idx] = result;
  updateToggles();
  updateLtmIndicator();

  // MtL: auto-show prescribed prompt for next context cell
  if (state.promptMethod === 'mtl') {
    displayPrompt('');
    autoShowForContext();
  }
}

// ─── Prompt Logic ─────────────────────────────────────────────────────────────

function pickNonRepeating(list, used) {
  let avail = list.map((_, i) => i).filter(i => !used.includes(i));
  if (avail.length === 0) { used.length = 0; avail = list.map((_, i) => i); }
  const pick = avail[Math.floor(Math.random() * avail.length)];
  used.push(pick);
  return list[pick];
}

// Returns the contextual next {turn, typeIdx} based on toggle state, or null if done
function getContextCell() {
  if (state.turns.some(t => !t[0])) {
    const turn = state.turns.findIndex(t => !t[0]);
    return { turn, typeIdx: 0 };
  }
  for (let t = 0; t < 3; t++) {
    if (!state.turns[t][1]) return { turn: t, typeIdx: 1 };
  }
  if (!state.turns[3][1]) return { turn: 3, typeIdx: 1 };
  return null;
}

// Returns prompt text for a specific cell at a given level
function getPromptForCell(turn, typeIdx, level) {
  const name       = state.currentPerson.name;
  const isComment  = typeIdx === 0;
  const isClosing  = !isComment && turn === 3;

  if (level === 0) return null; // independence — no prompt

  if (level === 1) {
    return isComment ? 'Comment: . . .' : 'Ask: . . .';
  }

  if (level === 2) {
    if (isClosing) {
      return pickNonRepeating(PARTIAL_CLOSING_PROMPTS, state.usedPartialClosingPrompts).replace('[name]', name);
    }
    if (isComment) {
      return pickNonRepeating(PARTIAL_COMMENT_PROMPTS, state.usedPartialCommentPrompts).replace('[name]', name);
    }
    return pickNonRepeating(PARTIAL_VOLLEY_PROMPTS, state.usedPartialVolleyPrompts).replace('[name]', name);
  }

  // level === 3 (full)
  if (isClosing) {
    return 'Answer: ' + CLOSING_PROMPTS[Math.floor(Math.random() * CLOSING_PROMPTS.length)].replace('[name]', name);
  }
  if (isComment) {
    return 'Answer: ' + pickNonRepeating(COMMENT_PROMPTS, state.usedCommentPrompts).replace('[name]', name);
  }
  return 'Ask: ' + pickNonRepeating(VOLLEY_PROMPTS, state.usedVolleyPrompts);
}

function displayPrompt(text) {
  const el = $('prompt-display');
  if (!text) { el.textContent = ''; return; }
  el.textContent = text.startsWith('🎉') ? text : '\u201C' + text + '\u201D';
}

// Update LtM step indicator to reflect the current context cell's level
function updateLtmIndicator() {
  const cell  = getContextCell();
  const level = cell ? state.ltmCellLevels[cell.turn][cell.typeIdx] : 0;
  document.querySelectorAll('.ltm-level-step').forEach(el => {
    el.classList.toggle('active', parseInt(el.dataset.step) === level);
  });
}

// MtL: auto-show prescribed prompt for next context cell (with optional 3s delay)
let autoShowTimeout = null;

function autoShowForContext() {
  clearTimeout(autoShowTimeout);
  const cell = getContextCell();
  if (!cell) { displayPrompt('🎉 All turns complete!'); return; }
  const show = () => displayPrompt(getPromptForCell(cell.turn, cell.typeIdx, state.mtlLevel));
  if (state.mtlDelay) {
    autoShowTimeout = setTimeout(show, 3000);
  } else {
    show();
  }
}

// Main Prompt button
function showPrompt() {
  const cell = getContextCell();
  if (!cell) { displayPrompt('🎉 All turns complete!'); return; }

  if (state.promptMethod === 'mtl') {
    // Override to Full for this cell
    state.promptClickedForCell[cell.turn][cell.typeIdx] = true;
    clearTimeout(autoShowTimeout);
    displayPrompt(getPromptForCell(cell.turn, cell.typeIdx, 3));
  } else {
    // LtM: advance this cell's level (stay at 3 but re-generate if already full)
    const cur   = state.ltmCellLevels[cell.turn][cell.typeIdx];
    const level = cur < 3 ? cur + 1 : 3;
    state.ltmCellLevels[cell.turn][cell.typeIdx] = level;
    updateLtmIndicator();
    displayPrompt(getPromptForCell(cell.turn, cell.typeIdx, level));
  }
}

// Shared level selector (MtL=prescribed level, LtM=ceiling)
function setMtlLevel(level) {
  state.mtlLevel = level;
  document.querySelectorAll('.level-seg-btn').forEach(btn => {
    btn.classList.toggle('active', parseInt(btn.dataset.level) === level);
  });
  if (state.promptMethod === 'mtl') autoShowForContext();
}

// Method radio switch
function setPromptMethod(method) {
  state.promptMethod = method;
  const isLtm = method === 'ltm';
  $('ltm-level-indicator').classList.toggle('hidden', !isLtm);
  $('mtl-level-selector').classList.toggle('hidden',  isLtm);
  $('mtl-delay-label').classList.toggle('hidden',     isLtm);
  if (!isLtm) {
    autoShowForContext();
  } else {
    clearTimeout(autoShowTimeout);
    displayPrompt('');
    updateLtmIndicator();
  }
}

// ─── Session Flow ─────────────────────────────────────────────────────────────

function pickPerson() {
  let avail = PEOPLE.map((_, i) => i).filter(i => !state.usedPeople.has(i));
  if (avail.length === 0) { state.usedPeople.clear(); avail = PEOPLE.map((_, i) => i); }
  const idx = avail[Math.floor(Math.random() * avail.length)];
  state.usedPeople.add(idx);
  return PEOPLE[idx];
}

function resetSession() {
  state.turns                = [[null, null], [null, null], [null, null], [null, null]];
  state.usedCommentPrompts        = [];
  state.usedVolleyPrompts         = [];
  state.usedPartialCommentPrompts = [];
  state.usedPartialVolleyPrompts  = [];
  state.usedPartialClosingPrompts = [];
  state.ltmCellLevels          = [[0, 0], [0, 0], [0, 0], [0, 0]];
  state.promptClickedForCell   = [[false, false], [false, false], [false, false], [false, false]];
  clearTimeout(autoShowTimeout);
  displayPrompt('');
  updateToggles();
  updateLtmIndicator();
}

function startSession() {
  state.currentPerson = personFromHash() || pickPerson();
  state.usedPeople.add(PEOPLE.indexOf(state.currentPerson));
  state.palette       = PALETTES[Math.floor(Math.random() * PALETTES.length)];
  applyPalette(state.palette);
  resetSession();
  renderPerson(state.currentPerson);
  showScreen('session');
}

function newPerson() {
  state.currentPerson = pickPerson();
  state.palette       = PALETTES[Math.floor(Math.random() * PALETTES.length)];
  applyPalette(state.palette);
  resetSession();
  renderPerson(state.currentPerson);
}

// ─── Timer ────────────────────────────────────────────────────────────────────

const RING_CIRC = 2 * Math.PI * 93;

function formatTime(secs) {
  return Math.floor(secs / 60) + ':' + String(secs % 60).padStart(2, '0');
}

function updateRing() {
  const offset = (1 - state.timerRemaining / state.timerTotal) * RING_CIRC;
  $('ring-fg').style.strokeDashoffset = offset.toFixed(2);
}

function tickPulse() {
  const el = $('timer-display');
  el.classList.remove('tick');
  void el.offsetWidth;
  el.classList.add('tick');
}

function timerTick() {
  if (state.timerRemaining <= 0) {
    clearInterval(state.timerInterval);
    state.timerInterval = null;
    state.timerRunning  = false;
    $('timer-display').textContent = '0:00';
    updateRing();
    playChime();
    showTimerDone();
    return;
  }
  state.timerRemaining--;
  $('timer-display').textContent = formatTime(state.timerRemaining);
  updateRing();
  tickPulse();
}

function startTimer() {
  if (state.timerRunning) return;
  state.timerRunning = true;
  $('btn-play-pause').textContent = '⏸ Pause';
  state.timerInterval = setInterval(timerTick, 1000);
}

function pauseTimer() {
  clearInterval(state.timerInterval);
  state.timerInterval = null;
  state.timerRunning  = false;
  $('btn-play-pause').textContent = '▶ Resume';
}

function stopTimer() {
  clearInterval(state.timerInterval);
  state.timerInterval = null;
  state.timerRunning  = false;
  showTimerDone();
}

function showTimerDone() {
  $('timer-controls').classList.add('hidden');
  $('timer-done').classList.remove('hidden');
}

function resetTimer() {
  clearInterval(state.timerInterval);
  state.timerInterval  = null;
  state.timerRunning   = false;
  state.timerRemaining = state.timerTotal;
  $('timer-display').textContent = formatTime(state.timerTotal);
  $('ring-fg').style.strokeDashoffset = '0';
  $('ring-fg').style.transition = 'none';
  requestAnimationFrame(() => {
    $('ring-fg').style.transition = '';
  });
  $('btn-play-pause').textContent = '▶ Start';
  $('timer-controls').classList.remove('hidden');
  $('timer-done').classList.add('hidden');
}

// ─── Chime ────────────────────────────────────────────────────────────────────

function playChime() {
  try {
    const ctx = new (window.AudioContext || window.webkitAudioContext)();
    [523, 659, 784].forEach((freq, i) => {
      const osc  = ctx.createOscillator();
      const gain = ctx.createGain();
      osc.connect(gain);
      gain.connect(ctx.destination);
      osc.type = 'sine';
      osc.frequency.value = freq;
      const t = ctx.currentTime + i * 0.32;
      gain.gain.setValueAtTime(0.35, t);
      gain.gain.exponentialRampToValueAtTime(0.001, t + 1.1);
      osc.start(t);
      osc.stop(t + 1.2);
    });
  } catch (_) { /* AudioContext unavailable */ }
}

// ─── Particles ────────────────────────────────────────────────────────────────

function createParticles() {
  const c = $('particles');
  c.innerHTML = '';
  for (let i = 0; i < 5; i++) {
    const p = document.createElement('div');
    p.className = 'particle';
    const size = 5 + Math.random() * 9;
    const left = 15 + Math.random() * 70;
    const dur  = 3.2 + Math.random() * 3;
    const del  = -(Math.random() * dur);
    const dft  = ((Math.random() - 0.5) * 50).toFixed(1);
    p.style.cssText =
      `width:${size}px;height:${size}px;` +
      `left:${left}%;bottom:8%;` +
      `animation-duration:${dur}s;` +
      `animation-delay:${del}s;` +
      `--drift:${dft}px;`;
    c.appendChild(p);
  }
}

function goToSR() {
  resetTimer();
  createParticles();
  showScreen('sr');
}

// ─── Hash-based person lookup ─────────────────────────────────────────────────

function personFromHash() {
  const hash = location.hash.slice(1).trim().toLowerCase();
  if (!hash) return null;
  return PEOPLE.find(p => p.name.toLowerCase().startsWith(hash)) || null;
}
